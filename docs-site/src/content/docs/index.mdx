import { Card, CardGrid, LinkCard } from '@astrojs/starlight/components';

These docs cover installing and using Keyhole. They stay intentionally short and
link back to the repository, which is the source of truth for release details,
verification, and contribution.

<CardGrid stagger>
	<Card title="Browser" icon="seti:db">
		Navigate the Redis keyspace and inspect values, with a live
		server-statistics band and a pinned, read-only command console.
	</Card>
	<Card title="Realtime" icon="random">
		Live tails of pub/sub, streams, keyspace events and `MONITOR` (Redis);
		topic/queue tails (AMQP 1.0); and exchange taps (RabbitMQ).
	</Card>
	<Card title="Recording" icon="document">
		Record any live tail to a lossless JSONL file, browsable in the in-app
		recordings viewer.
	</Card>
	<Card title="Observe-first" icon="approve-check">
		Browsing, inspecting, tailing and recording are all read-only. Secrets
		are never written to disk in plaintext.
	</Card>
</CardGrid>
